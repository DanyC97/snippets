snippet lib
    library(${1:package})

snippet req
    require(${1:package})

snippet src
    source("${1:file.R}")

snippet ret
    return(${1:code})

snippet mat
    matrix(${1:data}, nrow = ${2:rows}, ncol = ${3:cols})

snippet sg
    setGeneric("${1:generic}", function(${2:x, ...}) {
        standardGeneric("${1:generic}")
    })

snippet sm
    setMethod("${1:generic}", ${2:class}, function(${2:x, ...}) {
        ${0}
    })

snippet sc
    setClass("${1:Class}", slots = c(${2:name = "type"}))

snippet if
    if (${1:condition}) {
        ${0}
    }

snippet el
    else {
        ${0}
    }

snippet ei
    else if (${1:condition}) {
        ${0}
    }

snippet fun
    ${1:name} <- function(${2:variables}) {
        ${0}
    }

        ${0}
    }

snippet while

        ${0}
    }

# MLE (biased) estimate of variance
var.n = function(x) {
  v = var(x)
  n = NROW(x)
  v * (n - 1) / n
}
